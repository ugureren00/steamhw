```sql
CREATE TABLE developers (
    developer_id      integer GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name              varchar(150) NOT NULL UNIQUE,
    country_code      varchar(2) NOT NULL,
    founded_at        date,
    website           text,
    is_verified       boolean NOT NULL DEFAULT false,
    meta_json         jsonb NOT NULL DEFAULT '{}'::jsonb
);


CREATE TABLE users (
    user_id            integer GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    username           varchar(50) NOT NULL UNIQUE,
    email              varchar(100) NOT NULL UNIQUE,
    password           varchar(100) NOT NULL,
    registration_date  date NOT NULL DEFAULT CURRENT_DATE,
    phone              varchar(20),
    role               text NOT NULL DEFAULT 'user',
    country_code       varchar(2) NOT NULL DEFAULT 'US',
    account_status     text NOT NULL DEFAULT 'active',
    profile_bio        text,
    preferences_json   jsonb NOT NULL DEFAULT '{}'::jsonb,
    avatar_url         text,
    last_login_at      timestamp
);

CREATE TABLE games (
    game_id               integer GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    title                 varchar(200) NOT NULL,
    release_date          date,
    developer_id          integer REFERENCES developers(developer_id),
    avg_rating            numeric(4,2),
    review_count          integer NOT NULL DEFAULT 0,
    description           text,
    age_rating            text NOT NULL DEFAULT '16+',
    popularity_bucket     text NOT NULL DEFAULT 'mid',
    supported_languages   text[] NOT NULL DEFAULT ARRAY['en'],
    requirements_json     jsonb NOT NULL DEFAULT '{}'::jsonb,
    active_period         daterange
);

CREATE TABLE products (
    product_id       integer GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name             varchar(100) NOT NULL,
    price            numeric(10,2) NOT NULL CHECK (price >= 0),
    type             varchar(50),
    game_id          integer REFERENCES games(game_id),
    discount         numeric(5,2) NOT NULL DEFAULT 0 CHECK (discount >= 0 AND discount <= 100),
    description      text,
    tags             text[] NOT NULL DEFAULT ARRAY[]::text[],
    metadata_json    jsonb NOT NULL DEFAULT '{}'::jsonb,
    publisher_name   varchar(100) NOT NULL DEFAULT 'Unknown Publisher',
    sale_period      tsrange,
    is_active        boolean NOT NULL DEFAULT true,
    created_at       timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at       timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE orders (
    order_id         integer GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    user_id          integer NOT NULL REFERENCES users(user_id),
    order_date       timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
    product_id       integer REFERENCES products(product_id),
    status           text NOT NULL DEFAULT 'pending',
    final_price      numeric(10,2) NOT NULL DEFAULT 0,
    payment_method   text NOT NULL DEFAULT 'card',
    coupon_code      text,
    extra_data       jsonb NOT NULL DEFAULT '{}'::jsonb,
    refund_period    tsrange,
    currency_code    varchar(10) NOT NULL DEFAULT 'USD'
);

CREATE TABLE order_items (
    order_id            integer NOT NULL REFERENCES orders(order_id) ON DELETE CASCADE,
    product_id          integer NOT NULL REFERENCES products(product_id),
    qty                 integer NOT NULL DEFAULT 1 CHECK (qty > 0),
    price_at_purchase   numeric(10,2) NOT NULL CHECK (price_at_purchase >= 0),
    line_discount       numeric(10,2) NOT NULL DEFAULT 0 CHECK (line_discount >= 0),
    is_gift             boolean NOT NULL DEFAULT false,
    gift_wrap           boolean NOT NULL DEFAULT false,
    added_at            timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (order_id, product_id)
);

CREATE TABLE payments (
    payment_id        integer GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    order_id          integer NOT NULL REFERENCES orders(order_id),
    amount            numeric(10,2) NOT NULL CHECK (amount >= 0),
    method            varchar(50),
    status            varchar(30),
    paid_at           timestamp,
    provider_txn_id   varchar(120),
    processor_meta    jsonb NOT NULL DEFAULT '{}'::jsonb
);

CREATE TABLE libraries (
    user_id           integer NOT NULL REFERENCES users(user_id),
    product_id        integer NOT NULL REFERENCES products(product_id),
    acquired_at       timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
    source_type       text NOT NULL DEFAULT 'purchase',
    playtime_minutes  integer NOT NULL DEFAULT 0,
    last_played_at    timestamp,
    install_region    varchar(20),
    install_meta      jsonb NOT NULL DEFAULT '{}'::jsonb,
    PRIMARY KEY (user_id, product_id)
);

CREATE TABLE reviews (
    user_id             integer NOT NULL REFERENCES users(user_id),
    product_id          integer NOT NULL REFERENCES products(product_id),
    rating              integer NOT NULL CHECK (rating >= 1 AND rating <= 10),
    comment             text,
    review_date         timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
    edited_at           timestamp,
    language_code       varchar(10) NOT NULL DEFAULT 'en',
    is_recommended      boolean NOT NULL DEFAULT true,
    reaction_summary    jsonb NOT NULL DEFAULT '{}'::jsonb,
    visibility_status   text NOT NULL DEFAULT 'public',
    PRIMARY KEY (user_id, product_id)
);
```
